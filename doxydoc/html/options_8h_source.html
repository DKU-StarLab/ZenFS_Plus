54  // errors. This may have unforeseen ramifications: for example, a

71  // -------------------
72  // Parameters that affect performance
73 
74  // Amount of data to build up in memory (backed by an unsorted log
75  // on disk) before converting to a sorted on-disk file.
76  //
77  // Larger values increase performance, especially during bulk loads.
78  // Up to two write buffers may be held in memory at the same time,
79  // so you may wish to adjust this parameter to control memory usage.
80  // Also, a larger write buffer will result in a longer recovery time
81  // the next time the database is opened.
82  //

85

87  // increase this if your database has a large working set (budget
88  // one open file per 2MB of working set).
89  //

91  int max_open_files;

94  // a block is the unit of reading from disk).

97  // If null, leveldb will automatically create and use an 8MB internal cache.
98  // Default: nullptr

100

102  // block size specified here corresponds to uncompressed data. The
103  // actual size of the unit read from disk may be smaller if
104  // compression is enabled. This parameter can be changed dynamically.
105  //
106  // Default: 4K
107  size_t block_size;
108

112  //

115 
116  // Leveldb will write up to this amount of bytes to a file before
117  // switching to a new one.
118  // Most clients should leave this parameter alone. However if your
119  // filesystem is more efficient with larger files, you could
120  // consider increasing the value. The downside will be longer
121  // compactions and hence longer latency/performance hiccups.
122  // Another reason to increase this parameter might be when you are
123  // initially populating a large database.
124  //
125  // Default: 2MB
126  size_t max_file_size;
127 
128  // Compress blocks using the specified compression algorithm. This
129  // parameter can be changed dynamically.
130  //
131  // Default: kSnappyCompression, which gives lightweight but fast
132  // compression.
133  //
134  // Typical speeds of kSnappyCompression on an Intel(R) Core(TM)2 2.4GHz:

154  // Default: nullptr

187 // Options that control write operations
188 struct LEVELDB_EXPORT WriteOptions {
189  // If true, the write will be flushed from the operating system
190  // buffer cache (by calling WritableFile::Sync()) before the write
191  // is considered complete. If this flag is true, writes will be
192  // slower.
193  //
194  // If this flag is false, and the machine crashes, some recent
195  // writes may be lost. Note that if it is just the process that
196  // crashes (i.e., the machine does not reboot), no writes will be
197  // lost even if sync==false.
198  //
199  // In other words, a DB write with sync==false has similar
200  // crash semantics as the "write()" system call. A DB write
201  // with sync==true has similar crash semantics to a "write()"
202  // system call followed by "fsync()".
203  //
204  // Default: false
205  bool sync;
206 
207  WriteOptions()
208  : sync(false) {
209  }

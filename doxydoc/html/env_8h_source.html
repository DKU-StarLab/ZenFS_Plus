5 // An Env is an interface used by the leveldb implementation to access
6 // operating system functionality like the filesystem etc. Callers
7 // may wish to provide a custom Env object when opening a database to
8 // get fine gain control; e.g., to rate limit file system operations.
9 //
10 // All Env implementations are safe for concurrent access from
11 // multiple threads without any external synchronization.

41  // Return a default environment suitable for the current operating
42  // system. Sophisticated users may wish to provide their own Env
43  // implementation instead of relying on this default environment.
44  //
45  // The result of Default() belongs to leveldb and must never be deleted.

48  // Create an object that sequentially reads the file with the specified name.
49  // On success, stores a pointer to the new file in *result and returns OK.
50  // On failure stores nullptr in *result and returns non-OK. If the file does
51  // not exist, returns a non-OK status. Implementations should return a
52  // NotFound status when the file does not exist.
53  //
54  // The returned file will only be accessed by one thread at a time.
55  virtual Status NewSequentialFile(const std::string& fname,
56  SequentialFile** result) = 0;
57 
58  // Create an object supporting random-access reads from the file with the
59  // specified name. On success, stores a pointer to the new file in
60  // *result and returns OK. On failure stores nullptr in *result and
61  // returns non-OK. If the file does not exist, returns a non-OK
62  // status. Implementations should return a NotFound status when the file does
63  // not exist.
64  //
65  // The returned file may be concurrently accessed by multiple threads.
66  virtual Status NewRandomAccessFile(const std::string& fname,
67  RandomAccessFile** result) = 0;
68 
69  // Create an object that writes to a new file with the specified
70  // name. Deletes any existing file with the same name and creates a
71  // new file. On success, stores a pointer to the new file in
72  // *result and returns OK. On failure stores nullptr in *result and
73  // returns non-OK.
74  //
75  // The returned file will only be accessed by one thread at a time.
76  virtual Status NewWritableFile(const std::string& fname,
77  WritableFile** result) = 0;
78 
79  // Create an object that either appends to an existing file, or
80  // writes to a new file (if the file does not exist to begin with).
81  // On success, stores a pointer to the new file in *result and
82  // returns OK. On failure stores nullptr in *result and returns
83  // non-OK.
84  //
85  // The returned file will only be accessed by one thread at a time.
86  //
87  // May return an IsNotSupportedError error if this Env does
88  // not allow appending to an existing file. Users of Env (including
89  // the leveldb implementation) must be prepared to deal with
90  // an Env that does not support appending.
91  virtual Status NewAppendableFile(const std::string& fname,
92  WritableFile** result);

114

117  const std::string& target) = 0;

119  // Lock the specified file. Used to prevent concurrent access to
120  // the same db by multiple processes. On failure, stores nullptr in
121  // *lock and returns non-OK.
122  //
123  // On success, stores a pointer to the object that represents the
124  // acquired lock in *lock and returns OK. The caller should call
125  // UnlockFile(*lock) to release the lock. If the process exits,
126  // the lock will be automatically released.
127  //
128  // If somebody else already holds the lock, finishes immediately
129  // with a failure. I.e., this call does not wait for existing locks
130  // to go away.
131  //
132  // May create the named file if it does not already exist.
133  virtual Status LockFile(const std::string& fname, FileLock** lock) = 0;
134 
135  // Release the lock acquired by a previous successful call to LockFile.
136  // REQUIRES: lock was returned by a successful LockFile() call
137  // REQUIRES: lock has not already been unlocked.
138  virtual Status UnlockFile(FileLock* lock) = 0;
139 
140  // Arrange to run "(*function)(arg)" once in a background thread.
141  //
142  // "function" may run in an unspecified thread. Multiple functions
143  // added to the same Env may run concurrently in different threads.
144  // I.e., the caller may not assume that background work items are
145  // serialized.
146  virtual void Schedule(
147  void (*function)(void* arg),
148  void* arg) = 0;
149 
150  // Start a new thread, invoking "function(arg)" within the new thread.
151  // When "function(arg)" returns, the thread will be destroyed.
152  virtual void StartThread(void (*function)(void* arg), void* arg) = 0;
153 
154  // *path is set to a temporary directory that can be used for testing. It may
155  // or many not have just been created. The directory may or may not differ
156  // between runs of the same process, but subsequent calls will return the
157  // same directory.
158  virtual Status GetTestDirectory(std::string* path) = 0;
159 
160  // Create and return a log file for storing informational messages.
161  virtual Status NewLogger(const std::string& fname, Logger** result) = 0;
162 
163  // Returns the number of micro-seconds since some fixed point in time. Only
164  // useful for computing deltas of time.
165  virtual uint64_t NowMicros() = 0;
166 
167  // Sleep/delay the thread for the prescribed number of micro-seconds.
168  virtual void SleepForMicroseconds(int micros) = 0;
169 };

171 // A file abstraction for reading sequentially through a file
172 class LEVELDB_EXPORT SequentialFile {
173  public:
174  SequentialFile() = default;
175 
176  SequentialFile(const SequentialFile&) = delete;
177  SequentialFile& operator=(const SequentialFile&) = delete;
178 
179  virtual ~SequentialFile();
180 
181  // Read up to "n" bytes from the file. "scratch[0..n-1]" may be
182  // written by this routine. Sets "*result" to the data that was
183  // read (including if fewer than "n" bytes were successfully read).
184  // May set "*result" to point at data in "scratch[0..n-1]", so
185  // "scratch[0..n-1]" must be live when "*result" is used.
186  // If an error was encountered, returns a non-OK status.
187  //
188  // REQUIRES: External synchronization
189  virtual Status Read(size_t n, Slice* result, char* scratch) = 0;
190 
191  // Skip "n" bytes from the file. This is guaranteed to be no
192  // slower that reading the same data, but may be faster.
193  //
194  // If end of file is reached, skipping will stop at the end of the
195  // file, and Skip will return OK.
196  //
197  // REQUIRES: External synchronization
198  virtual Status Skip(uint64_t n) = 0;
199 };

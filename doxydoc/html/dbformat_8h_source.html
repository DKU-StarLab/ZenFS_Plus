166 inline bool ParseInternalKey(const Slice& internal_key,
167  ParsedInternalKey* result) {
168  const size_t n = internal_key.size();
169  if (n < 8) return false;
170  uint64_t num = DecodeFixed64(internal_key.data() + n - 8);
171  unsigned char c = num & 0xff;
172  result->sequence = num >> 8;
173  result->type = static_cast<ValueType>(c);
174  result->user_key = Slice(internal_key.data(), n - 8);
175  return (c <= static_cast<unsigned char>(kTypeValue));
176 }
